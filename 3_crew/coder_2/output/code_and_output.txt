# Calculate the first 10,000 terms of the series
def calculate_series(terms):
    total = 0
    for i in range(terms):
        # Compute the current term, using the formula
        if i % 2 == 0:  # Even index: Add
            total += 1 / (2 * i + 1)
        else:           # Odd index: Subtract
            total -= 1 / (2 * i + 1)
    
    # Multiply total by 4
    total *= 4
    return total

# Set the number of terms
num_terms = 10000
result = calculate_series(num_terms)
print(result)
